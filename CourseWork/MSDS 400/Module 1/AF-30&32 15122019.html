#Set up least squares functions
def least_squares_m(x,y):
    n = len(x)
    xy = []
    for i in range(len(x)):
        xy = xy + [x[i]*y[i]]
    x_sq = []
    for i in range(len(x)):
        x_sq = x_sq + [x[i]**2]
    sum_xy = sum(xy)
    sum_x = sum(x)
    sum_y = sum(y)
    sum_x_sq = sum(x_sq)
    m = ((n*sum_xy)-(sum_x*sum_y))/((n*sum_x_sq)-(sum_x**2))
    return m
def least_squares_b(x,y,m):
    sum_y = sum(y)
    sum_x = sum(x)
    n = len(x)
    b = (sum_y - (m*sum_x))/n
    return b
